Model evaluation with cross-validation [!]

Model evaluation using different metrics computed with a test dataset.

records <- read.csv("Data/dataBases/Paramuricea_clavata.csv",sep=";")
records <- records[,c("Lon","Lat")]

# load layers of sea surface temperatures and nutrients
environmentalConditionsPresent <- load_layers(c("BO2_tempmin_bdmean","BO2_tempmax_bdmean"))

# crop layers to the European extent
europeanExtent <- extent(-15, 40, 25, 50)
environmentalConditionsPresent <- crop(environmentalConditionsPresent,europeanExtent)

# generate background information
background <- backgroundInformation(environmentalConditionsPresent,n=10000)

# extract environmental values and make a data.frame with PA information
modelData <- prepareSWD(species = "P Clavata", p = records, a = background, env = environmentalConditionsPresent)

# Cross validation with random partitioning of 10 folds
folds <- randomFolds(modelData, k = 10)

# fit a Maxent model to the dataset
model <- train("Maxnet", modelData, folds = folds )

# determine AUC
getAUC(model)
## [1] 0.9210017
getAUC(model, test = TRUE)
## [1] 0.9171685
# Cross validation with 4 blocks partitioning
folds2 <- get.block(records, background)

# fit a Maxent model to the dataset
model <- train("Maxnet", modelData, folds = folds2 )

# determine AUC
getAUC(model)
## [1] 0.9256193
getAUC(model, test = TRUE)
## [1] 0.9139369
